
	\subsection{ویژگی‌های استخراج‌شده از هر سیگنال}
	\label{app:features}
		در این پژوهش، برای هر کانال از سیگنال‌های شتاب‌سنج و ژیروسکوپ، مجموعه‌ای از ویژگی‌های حوزه زمان استخراج شد. اگر سیگنال ورودی به صورت
		$x = \{x_1, x_2, \ldots, x_N\}$	در نظر گرفته شود، که در آن $N$ تعداد نمونه‌های سیگنال است، ویژگی‌های استخراج‌شده مطابق جدول
		\ref{tab:features}	تعریف شدند.
		در روابط زیر، سیگنال ورودی به صورت
	$x = \{x_1, x_2, \ldots, x_N\}$	در نظر گرفته شده است. همچنین $\mu$ میانگین سیگنال، $\sigma$ انحراف معیار، $Q_1$ چارک اول، $Q_3$ چارک سوم و
		$\Delta x_i = x_{i+1}-x_i$	تفاضل نمونه‌های متوالی را نشان می‌دهد.

			1
			&			تعداد نمونه‌ها
				&
			\lr{Count}
			&			تعداد کل نمونه‌های موجود در سیگنال
				&
			$N$
			\\
			
			2
			&			میانگین
				&
			\lr{Mean}
			&			مقدار متوسط سیگنال
				&
			$\mu = \dfrac{1}{N}\displaystyle\sum_{i=1}^{N}x_i$
			\\
			
			3
			&			انحراف معیار
				&
			\lr{Standard Deviation}
			&			میزان پراکندگی نمونه‌ها نسبت به میانگین
				&
			$\sigma =
			\sqrt{
				\dfrac{1}{N-1}
				\displaystyle\sum_{i=1}^{N}(x_i-\mu)^2
			}$
			\\
			
			4
			&			واریانس
				&
			\lr{Variance}
			&			شاخص پراکندگی سیگنال
				&
			$s^2 =
			\dfrac{1}{N-1}
			\displaystyle\sum_{i=1}^{N}(x_i-\mu)^2$
			\\
			
			5
			&			کمینه
				&
			\lr{Minimum}
			&			کمترین مقدار ثبت‌شده در سیگنال
				&
			$\min(x)=\min\{x_1,\ldots,x_N\}$
			\\
			
			6
			&			بیشینه
				&
			\lr{Maximum}
			&			بیشترین مقدار ثبت‌شده در سیگنال
				&
			$\max(x)=\max\{x_1,\ldots,x_N\}$
			\\
			
			7
			&			دامنه تغییرات
				&
			\lr{Range}
			&			اختلاف بین بیشینه و کمینه سیگنال
				&
			$Range=\max(x)-\min(x)$
			\\
			
			8
			&			میانه
				&
			\lr{Median}
			&			مقدار مرکزی سیگنال پس از مرتب‌سازی نمونه‌ها
				&
			$\widetilde{x}=median(x)$
			\\
			
			9
			&			چارک اول
				&
			\lr{First Quartile}
			&			صدک ۲۵ام سیگنال
				&
			$Q_1=P_{25}(x)$
			\\
			
			10
			&			چارک سوم
				&
			\lr{Third Quartile}
			&			صدک ۷۵ام سیگنال
				&
			$Q_3=P_{75}(x)$
			\\
			
			11
			&			دامنه بین‌چارکی
				&
			\lr{Interquartile Range}
			&			اختلاف بین چارک سوم و چارک اول
				&
			$IQR=Q_3-Q_1$
			\\
			
			12
			&			میانگین قدر مطلق انحرافات
				&
			\lr{Mean Absolute Deviation}
			&			میانگین فاصله قدر مطلق نمونه‌ها از میانگین
				&
			$MAD=
			\dfrac{1}{N}
			\displaystyle\sum_{i=1}^{N}|x_i-\mu|$
			\\
			
			13
			&			میانگین قدر مطلق
				&
			\lr{Mean Absolute Value}
			&			میانگین اندازه مطلق مقادیر سیگنال
				&
			$MeanAbs=
			\dfrac{1}{N}
			\displaystyle\sum_{i=1}^{N}|x_i|$
			\\
			
			14
			&			ریشه میانگین مربعات
				&
			\lr{Root Mean Square}
			&			شاخصی از شدت کلی سیگنال
				&
			$RMS=
			\sqrt{
				\dfrac{1}{N}
				\displaystyle\sum_{i=1}^{N}x_i^2
			}$
			\\
			
			15
			&			انرژی سیگنال
				&
			\lr{Energy}
			&			مجموع توان نمونه‌های سیگنال
				&
			$Energy=
			\displaystyle\sum_{i=1}^{N}x_i^2$
			\\
			
			16
			&			ضریب تغییرات
				&
			\lr{Coefficient of Variation}
			&			نسبت انحراف معیار به میانگین
				&
			$CV=\dfrac{\sigma}{|\mu|}$
			\\
			
			17
			&			چولگی
				&
			\lr{Skewness}
			&			شاخص عدم تقارن توزیع مقادیر سیگنال
				&
			$Skewness=
			\dfrac{1}{N}
			\displaystyle\sum_{i=1}^{N}
			\left(
			\dfrac{x_i-\mu}{\sigma}
			\right)^3$
			\\
			
			18
			&			کشیدگی
				&
			\lr{Kurtosis}
			&			شاخص میزان تیزی یا پهنی توزیع سیگنال
				&
			$Kurtosis=
			\dfrac{1}{N}
			\displaystyle\sum_{i=1}^{N}
			\left(
			\dfrac{x_i-\mu}{\sigma}
			\right)^4$
			\\
			
			19
			&			آنتروپی
				&
			\lr{Entropy}
			&			شاخصی از میزان بی‌نظمی یا پراکندگی توزیع سیگنال
				&
			$Entropy=
			-\displaystyle\sum_{j=1}^{B}p_j\log(p_j)$
			\\
			
			20
			&			تعداد عبور از صفر
				&
			\lr{Zero Crossings}
			&			تعداد دفعات تغییر علامت سیگنال
				&
			$ZC=
			\displaystyle\sum_{i=1}^{N-1}
			I(x_i x_{i+1}<0)$
			\\
			
			21
			&			میانگین تفاضل نمونه‌های متوالی
				&
			\lr{Mean of First Differences}
			&			میانگین تغییرات بین نمونه‌های متوالی
				&
			$DiffMean=
			\dfrac{1}{N-1}
			\displaystyle\sum_{i=1}^{N-1}
			(x_{i+1}-x_i)$
			\\
			
			22
			&			انحراف معیار تفاضل نمونه‌های متوالی
				&
			\lr{Standard Deviation of First Differences}
			&			میزان پراکندگی تغییرات متوالی سیگنال
				&
			$DiffStd=std(\Delta x)$
			\\
			
			23
			&			خودهمبستگی با تأخیر یک
				&
			\lr{Lag-1 Autocorrelation}
			&			میزان وابستگی هر نمونه به نمونه بعدی
				&
			$\displaystyle
			Autocorr_1=
			\dfrac{
				\sum_{i=1}^{N-1}
				(x_i-\mu)(x_{i+1}-\mu)
			}{
				\sum_{i=1}^{N}
				(x_i-\mu)^2
			}$
			\\
		\end{longtable}
